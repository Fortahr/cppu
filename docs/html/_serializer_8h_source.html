441  template<typename T>
442  inline T SwapEndian(T value)
443  {
444  // code thanks to Google FlatBuffers
445 # if defined(_MSC_VER)
446 # define PLATFORM_BYTE_SWAP_16 _byteswap_ushort
447 # define PLATFORM_BYTE_SWAP_32 _byteswap_ulong
448 # define PLATFORM_BYTE_SWAP_64 _byteswap_uint64
449 # elif defined(__ICCARM__)
450 # define PLATFORM_BYTE_SWAP_16 __REV16
451 # define PLATFORM_BYTE_SWAP_32 __REV
452 # define PLATFORM_BYTE_SWAP_64(x) \
453  ((__REV(static_cast<uint32_t>(x >> 32U))) | (static_cast<uint64_t>(__REV(static_cast<uint32_t>(x)))) << 32U)
454 # else
455 # if defined(__GNUC__) && __GNUC__ * 100 + __GNUC_MINOR__ < 408 && !defined(__clang__)
456  // __builtin_bswap16 was missing prior to GCC 4.8.
457 # define PLATFORM_BYTE_SWAP_16(x) static_cast<uint16_t>(__builtin_bswap32(static_cast<uint32_t>(x) << 16))
458 # else
459 # define PLATFORM_BYTE_SWAP_16 __builtin_bswap16
460 # endif
461 # define PLATFORM_BYTE_SWAP_32 __builtin_bswap32
462 # define PLATFORM_BYTE_SWAP_64 __builtin_bswap64
463 # endif
464  if constexpr (sizeof(T) == 1) // Compile-time if-then's.
465  return value;
466  else if constexpr (sizeof(T) == 2)
467  reinterpret_cast<uint16_t&>(value) = PLATFORM_BYTE_SWAP_16(reinterpret_cast<uint16_t&>(value));
468  else if constexpr (sizeof(T) == 4)
469  reinterpret_cast<uint32_t&>(value) = PLATFORM_BYTE_SWAP_32(reinterpret_cast<uint32_t&>(value));
470  else if constexpr (sizeof(T) == 8)
471  reinterpret_cast<uint64_t&>(value) = PLATFORM_BYTE_SWAP_64(reinterpret_cast<uint64_t&>(value));
472  //else
473  // static_assert(0);
474 
475  return value;
476  }
