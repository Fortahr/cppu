180  // shared pointer mechanics
181  void DecrementStrongReference()
182  {
183  auto* pointer = this->pointer;
184  auto* refCounter = this->refCounter;
185 
186  if (refCounter != nullptr)
187  {
188  // we'll check if the previous value was 2, if so it will decrement to 0 in this procedure then mark this one as garbage
189  // the last strong value holder is merely a safety measure so weak ptrs won't destroy the ref counter before we do so in here.
190  if (refCounter->strongReferences.fetch_sub(1) == 2)
191  {
192  // add it so the collection can clean it up and give out the free slot again
193  refCounter->add_as_garbage((void*)pointer);
194 
195  // after this we've set the actual strong ref count so any weak ptr can delete the ref counter from this point
196  refCounter->strongReferences.fetch_sub(1);
197 
198  // strong ref base_counter will be 0, now we only need to check the weak ref base_counter
199  // and remove the tracking object if it's 1
200  if (refCounter->weakReferences.load() == 0)
201  delete refCounter;
202  }
203  }
